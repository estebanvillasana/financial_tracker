Usage

Import and call render() with a target element
accountsSummary.js ES Module 
import { AccountsSummary } from './components/smart/accountsSummary/accountsSummary.js';

// Minimal usage — falls back to the runtime main currency
await AccountsSummary.render('#my-container');

// Full options
await AccountsSummary.render('#my-container', {
  pageSize:        5,           // cards per page (default: 5)
  columns:         2,           // grid columns: 1 or 2 (default: 2)
  defaultCurrency: 'USD',      // target for FX conversion labels
  title:           'Active Accounts',
});
